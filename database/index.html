数据库设计

CREATE TABLE t_user (
    userId INT auto_increment PRIMARY KEY,
    userName VARCHAR (255),
    userPass VARCHAR (255),
    userRole VARCHAR (255),
    nickleName VARCHAR (255),
    created DATETIME,
    lastUpdated DATETIME
);

CREATE TABLE t_loggedInUser (
    userId INT,
    userName VARCHAR (255),
    userRole VARCHAR (255),
    apiKey VARCHAR (255),
    created DATETIME,
    lastUpdated DATETIME
);

CREATE TABLE t_message (
    messageId INT auto_increment PRIMARY KEY,
    messageType VARCHAR (255),
    messageText VARCHAR (255),
    fromUserId INT,
    toUserId INT,
    bridgeId INT,
    bridgeName VARCHAR (255),
    flightNo VARCHAR (255),
    created DATETIME
);

CREATE TABLE t_userBridge (
    userId INT,
    userName VARCHAR (255),
    bridgeId INT,
    bridgeName VARCHAR (255),
    created DATETIME,
    lastUpdated DATETIME
);

CREATE TABLE t_bridge (
    bridgeId INT auto_increment PRIMARY KEY,
    bridgeName VARCHAR (255),
    bridgePosX DOUBLE,
    bridgePosY DOUBLE,
    bridgeStatus VARCHAR (255),
    created DATETIME,
    lastUpdated DATETIME
);
